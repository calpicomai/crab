# Perception (Jetson) dependencies — layered.
#
# EASIEST: run `bash brain/setup_perception.sh` — it installs the base subset
# below PLUS the Jetson-correct OpenCV (apt, with GStreamer) and Ultralytics, and
# guides the NVIDIA torch wheel. Prefer it over pip'ing this file by hand.
#
# Install this file's base subset alone with:
#   brain/.venv/bin/pip install -r brain/requirements-perception.txt
# The base subset enables the SERVER, the simulate/dummy path, and the test
# runner (so you can validate wiring before the heavy models are present).
#
# NOTE: torch and opencv are deliberately NOT listed as pip requirements — on
# Jetson the PyPI wheels are the wrong builds (torch: no CUDA; opencv-python: no
# GStreamer, so the CSI camera won't open). setup_perception.sh installs the
# correct ones.

# --- Base (pip-installable anywhere) -----------------------------------------
fastapi>=0.110
uvicorn[standard]>=0.29
pydantic>=2.6
# numpy < 2 on purpose: the JetPack stack (Jetson torch/torchvision, system
# OpenCV, matplotlib) is compiled against the numpy 1.x ABI. numpy 2 triggers
# "_ARRAY_API not found" in those C extensions (e.g. ultralytics importing
# matplotlib). Pinning <2 in the venv shadows any numpy 2 in ~/.local.
numpy>=1.24,<2
Pillow>=10.0

# --- JetPack / CUDA-specific — DO NOT plain-pip these; follow NVIDIA's steps ---
#
# OpenCV with GStreamer (for the CSI camera) is usually provided by JetPack.
# Verify GStreamer support:
#   python3 -c "import cv2; print('GStreamer:', 'YES' in cv2.getBuildInformation().split('GStreamer')[1][:20])"
#
# YOLO backend:
#   1) Install the NVIDIA Jetson torch + torchvision wheels for your JetPack.
#   2) pip install ultralytics        # sees the pre-installed torch
#   Default weights 'yolov8n.pt' download on first run; export to TensorRT for
#   speed:  yolo export model=yolov8n.pt format=engine  (then set PERCEPTION_YOLO_WEIGHTS=yolov8n.engine)
#
# NanoOWL backend (open-vocabulary):
#   `bash brain/setup_perception.sh --nanoowl` does all of this for you (it checks
#   for TensorRT first). By hand it needs torch, TensorRT (JetPack), transformers<5,
#   onnx + onnxscript (torch.onnx.export uses them), then torch2trt + nanoowl:
#     pip install "transformers<5" onnx onnxscript
#     git clone https://github.com/NVIDIA-AI-IOT/torch2trt && cd torch2trt && pip install .
#     git clone https://github.com/NVIDIA-AI-IOT/nanoowl  && cd nanoowl  && pip install .
#   Build the image-encoder TensorRT engine (path -> PERCEPTION_NANOOWL_ENGINE):
#     python3 -m nanoowl.build_image_encoder_engine data/owl_image_encoder_patch32.engine
